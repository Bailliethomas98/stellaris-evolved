triggered_planet_modifier = {
	potential = {
		planet = {
			exists = owner
			
			$trigger$

			OR = { 
				NOT = { exists = assembling_species }
				assembling_species = {
					is_robotic = yes
				}
			}
			
		}
	}
	modifier = {
		planet_pop_assembly_add = 1
	}
	mult = $mult$
}
