triggered_planet_modifier = {
	potential = {
		planet = {
			exists = owner

			$trigger$

			OR  = {
				NOT = { exists = assembling_species }
				can_assemble_hive_pop = yes
			}
		}
	}
	modifier = {
		planet_pop_assembly_organic_add = 1
	}
	mult = $mult$
}

	potential = {
		planet = {
			exists = owner

			$trigger$

			exists = assembling_species
			can_assemble_hive_pop = no
		}
	}
	tec_assembly_prohibited_no_hive = 1
}
