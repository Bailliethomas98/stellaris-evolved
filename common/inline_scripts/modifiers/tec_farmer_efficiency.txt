# modifiers/tec_farmer_efficiency

triggered_planet_modifier = {
	potential = {
		exists = owner

		$trigger$

		OR = {
			NOT = { exists = planet }
			planet = {
				tec_has_farmer_production_planet_swap = no
			}
		}

		owner = {
			tec_has_farmer_production_swap = no
		}
	}
	planet_farmers_food_produces_add = $count$
	mult = $mult$
}

inline_script = {
	script = mod_support/tec_inlines_include
	include_script = modifiers/tec_farmer_efficiency
	parameters = "
		count = $count$
		mult = $mult$
		trigger = \"$trigger$\"
	"
}
