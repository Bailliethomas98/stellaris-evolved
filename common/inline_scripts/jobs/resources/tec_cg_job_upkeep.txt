# jobs/resources/tec_cg_job_upkeep

upkeep = {
	trigger = {
		exists = owner

		$trigger$

		owner = {
			OR = {
				has_policy_flag = consumer_goods_minerals
				NOR = {
					has_policy_flag = consumer_goods_minerals
					has_policy_flag = consumer_goods_mixed
					has_policy_flag = consumer_goods_food
				}
			}
		}
	}
	minerals = $upkeep$
	multiplier = $mult$
}
upkeep = {
	trigger = {
		exists = owner

		$trigger$

		owner = {
			has_policy_flag = consumer_goods_mixed
		}
	}
	minerals = @[ $upkeep$ / 2]
	food = @[ $upkeep$ * 4 / 3 / 2]
	multiplier = $mult$
}
upkeep = {
	trigger = {
		exists = owner

		$trigger$

		owner = {
			has_policy_flag = consumer_goods_food
		}
	}
	food = @[ $upkeep$ * 4 / 3 ]
	multiplier = $mult$
}
