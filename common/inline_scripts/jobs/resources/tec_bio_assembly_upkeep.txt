# jobs/resources/tec_bio_assembly_upkeep

upkeep = {
	trigger = {
		planet = {
			exists = owner

			$trigger$

			OR = {
				AND = {
					NOT = { exists = assembling_species }
					owner = { is_lithoid = no }
				}
				AND = {
					exists = assembling_species
					assembling_species = {
						is_lithoid = no
					}
				}
			}
		}
	}
	food = $base$
	mult = $mult$
}
