# jobs/resources/tec_robotic_assembly_upkeep

upkeep = {
	trigger = {
		planet = {
			exists = owner
			
			$trigger$

			OR = {
				NOT = { exists = assembling_species }
				owner = { tec_allows_machine_assembly_for_cyborg = no }
				assembling_species = {
					tec_is_cyborg_species = no
				}
			}
		}
	}
	alloys = $base$
	mult = $mult$
}
upkeep = {
	trigger = {
		planet = {
			exists = owner
			exists = assembling_species

			$trigger$

			owner = { tec_allows_machine_assembly_for_cyborg = yes }
			assembling_species = {
				tec_is_cyborg_species = yes
				is_lithoid = no
			}
		}
	}
	alloys = @[$base$ / 2]
	food = @[$base$ / 2 * 5]
	mult = $mult$
}
upkeep = {

			exists = assembling_species

				tec_is_cyborg_species = yes

			}
		}
	}
	alloys = @[$base$ / 2]
	minerals = @[$base$ / 2 * 5]
	mult = $mult$
}
