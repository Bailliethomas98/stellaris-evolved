# tec_maintenance_civic_hive_swaps

# Hive maintenance swaps from civics on capital, useful for PW patches, 
# PD arcologies and other mods that replicate capital job behaviour

inline_script = { # Pherodrones
	script = jobs/tec_job
	trigger = "
		exists = owner

		$trigger$
		
		owner = { 
			has_valid_civic = civic_tec_hive_necrophage 
		}
	"
	job = tec_pherodrone_drone
	count = $count$
	mult = $mult$
	display = $display$
}

inline_script = {
	script = mod_support/tec_inlines_include
	include_script = jobs/tec_maintenance_civic_hive_swaps
	parameters = "
		count = $count$
		mult = $mult$
		display = $display$
		trigger = \"$trigger$\"
	"
}
