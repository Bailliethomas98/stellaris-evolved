# buildings/tec_aibait_essential


inline_script = {
	script = tec_type_hint
	trigger_block = "$trigger$"
}

ai_resource_production = {
	trigger = {
		exists = owner
		owner = {
			if = {
				limit = {
					tec_is_cyberarch_empire = yes
				}
				resource_stockpile_compare = {
					resource = minerals
					value >= @[tec_minimum_resources / 2]
				}
				resource_stockpile_compare = {
					resource = energy
					value >= @[tec_minimum_resources / 2]
				}
			}
			else_if = {
				limit = {
					tec_is_bioarch_empire = yes
					tec_is_biohive = yes
				}
				resource_stockpile_compare = {
					resource = food
					value >= @tec_minimum_resources
				}
			}
			else_if = {
				limit = {
					OR = {
						tec_is_bioarch_empire = yes
						tec_is_biohive = yes
					}
				}
				resource_stockpile_compare = {
					resource = minerals
					value >= @[tec_minimum_resources / 2]
				}
				resource_stockpile_compare = {
					resource = food
					value >= @[tec_minimum_resources / 2]
				}
			}
			else = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @tec_minimum_resources
				}
			}
		}
		$trigger$
	}
	ai_unobtainium = 2500
}
