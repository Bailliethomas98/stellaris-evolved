# buildings/tec_aibait_upgrade

ai_resource_production = {
	trigger = {
		exists = owner
		owner = {
			if = {
				limit = {
					tec_is_cyberarch_empire = yes
				}
				resource_stockpile_compare = {
					resource = minerals
					value >= @[tec_minimum_resources / 2]
				}
				resource_stockpile_compare = {
					resource = energy
					value >= @[tec_minimum_resources / 2]
				}
			}
			else_if = {
				limit = {
					OR = {
						tec_is_bioarch_empire = yes
						tec_is_biohive = yes
					}
				}
				resource_stockpile_compare = {
					resource = minerals
					value >= @[tec_minimum_resources / 2]
				}
				resource_stockpile_compare = {
					resource = food
					value >= @[tec_minimum_resources / 2]
				}
			}
			else_if = {
				limit = {
					tec_is_bioarch_empire = yes
					tec_is_biohive = yes
				}
				resource_stockpile_compare = {
					resource = food
					value >= @tec_minimum_resources
				}
			}
			else = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @tec_minimum_resources
				}
			}
		}
		owner = {
			NOR = {
				has_deficit = minerals
				has_deficit = energy
				has_deficit = food
				has_deficit = consumer_goods
			}
		}
		num_unemployed > 0
		OR = {
			AND = {
				free_building_slots = 0
				free_district_slots = 0
			}
			num_unemployed > 5
		}
		$trigger$
		any_owned_pop = {
			is_unemployed = yes
			can_work_specific_job = $job$
		}
	}
	ai_unobtainium = 1000
}
