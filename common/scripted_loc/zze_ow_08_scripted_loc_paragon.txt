defined_text = {
    name = GetTraitKeyPostfix
    random = no
    text = {
        localization_key = TRAIT_KEY_POSTFIX_MACHINE
        trigger = {
            is_scope_type = leader
            exists = species
            if = {
                limit = {
                    exists = creator
                }
                creator = {
                    is_machine_empire = yes
                }
            }
            else = {
                is_robotic_species = yes
            }
        }
    }
    text = {
        localization_key = TRAIT_KEY_POSTFIX_HIVE
        trigger = {
            is_scope_type = leader
            exists = species
            if = {
                limit = {
                    exists = creator
                }
                creator = {
                    is_hive_empire = yes
                }
            }
            else = {
                is_hive_species = yes
            }
        }
    }
    text = {
        trigger = {
            is_scope_valid = no
        }
        localization_key = TRAIT_KEY_POSTFIX_NONE
    }
    default = TRAIT_KEY_POSTFIX_NONE
}
