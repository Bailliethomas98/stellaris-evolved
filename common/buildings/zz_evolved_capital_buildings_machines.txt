		text = tec_machine_production_capital_effects
	}

		planet_housing_add = 5
		planet_amenities_add = 8
		planet_jobs_complex_drone_produces_mult = 0.1
		planet_jobs_complex_drone_upkeep_mult = 0.1

	# Biotrophies

		potential = {
			exists = owner

			}
		}
		planet_bio_trophies_produces_mult = 0.1
	}

	inline_script = {
		script = capitals/tec_all_capital
		tier = 1
	}

	inline_script = capitals/tec_colony
	inline_script = {
		script = capitals/tec_common_modifiers
		tier = 1
	}
	inline_script = {
		script = capitals/tec_machine_capital
		tier = 1
		ruler_trigger = ""
	}


	inline_script = {
		script = capitals/effects/tec_machine_assembly
		tier = 1
	}

	triggered_desc = {
		text = building_tec_capital_regular_production_tag
	}
	triggered_desc = {
