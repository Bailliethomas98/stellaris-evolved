# Resort Administration
building_resort_capital = {
	position_priority = 0		# Capital
	capital = yes
	can_build = no
	can_demolish = no
	can_be_ruined = no
	can_be_disabled = no
	icon = building_capital

	category = government

	potential = {
		has_modifier = resort_colony
	}

	upgrades = {
		"building_resort_major_capital"
	}

	inline_script = buildings/convert_to/tec_capital_t2

	resources = {
		category = planet_buildings
		upkeep = {
			energy = 2
		}
	}

	triggered_desc = {
		text = building_tec_capital_special_tag
	}
	triggered_desc = {
		text = building_tec_capital_tier_2_tag
	}

	planet_modifier = {
		planet_housing_add = 5
		planet_amenities_add = 5
		planet_max_buildings_add = @tec_capital_buildings_t2
		planet_max_branch_office_buildings_add = @tec_capital_branch_t1
	}

	inline_script = {
		script = capitals/tec_all_capital
		tier = 2
	}

	inline_script = {
		script = capitals/tec_common_modifiers
		tier = 2
	}

	inline_script = {
		script = capitals/tec_special_capital
		tier = 2
		count = 1
	}

	# Jobs
	inline_script = {
		script = jobs/tec_job
		job = resort_worker
		count = 1
		mult = 1
		display = with_desc
		trigger = "
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		"
	}
}
