# --------------- Habitation
district_arcology_housing = {
	base_buildtime = 600
	is_capped_by_modifier = no
	exempt_from_ai_planet_specialization = yes

	show_on_uncolonized = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		tec_is_decadent_city_planet = no
	}

	potential = {
		OR = {
			uses_district_set = city_world
			tec_pd_arcology_housing_trigger = yes
		}
		tec_is_decadent_city_planet = no
	}

	allow = {
		tec_ai_allow_housing_district = {
			housing = 5
			has_building_slots = "always = no"
			use_for_jobs = "
				tec_ai_allow_city_district_for_jobs = yes
			"
		}
	}
	
	conversion_ratio = 1.5
	inline_script = districts/convert_to/tec_housing_districts

	resources = {
		category = planet_districts_cities

		inline_script = {
			script = districts/resources/tec_district_resource_cost
			upkeep_trigger = "
				always = yes
			"
			cost = @tec_district_arc_cost
			upkeep = @tec_district_arc_upkeep
		}
	}

	# Modifier
		planet_modifier = {
			planet_housing_add = 15
		}
	# /
	# Effect Types
		inline_script = {
			script = districts/type/tec_housing_generic
			scale = 3
			trigger = ""
		}

		inline_script = {
			script = districts/type/tec_housing_arcology
			scale = 3
			trigger = ""
		}
	# /
	
	# Jobs
		# Regular
			inline_script = {
				script = jobs/tec_clerk
				count = 3
				mult = 1
				display = with_desc
				trigger = ""
			}
		# /
		# Gestalt
			inline_script = {
				script = jobs/tec_maintenance_drone
				count = 5
				mult = 1
				display = with_desc
				trigger = ""
			}
		# /
	# /


	inline_script = {
		script = districts/tec_aibait_housing
		has_building_slots = "
			always = no
		"
	}
}

district_arcology_civilian_industry = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = city_world
		OR = {
			tec_is_admin_city_planet = yes
			tec_is_unspecced_city_planet = yes
		}
		tec_is_decadent_city_planet = no
	}

	potential = {
		uses_district_set = city_world
		OR = {
			tec_is_admin_city_planet = yes
			tec_is_unspecced_city_planet = yes
		}
		tec_is_decadent_city_planet = no
	}

	conversion_ratio = 1
	inline_script = districts/convert_to/tec_ecu_cg_districts_swaps
	inline_script = districts/convert_to/tec_industrial_districts

	triggered_desc = {
		trigger = {
			NOT = {
				has_planet_flag = tec_ecumenopolis_spec_admin
			}
		}
		text = tec_useless_without_upgrading_planet_jobs
	}

	resources = {
		category = planet_districts_industrial

		inline_script = {
			script = districts/resources/tec_district_resource_cost
			upkeep_trigger = "
				always = yes
			"
			cost = @tec_district_arc_cost
			upkeep = @tec_district_arc_upkeep
		}
		inline_script = {
			script = districts/resources/tec_district_rare_resource_cost
			rare_upkeep_trigger = "
				always = yes
			"
			rare = rare_crystals
			rare_cost = @tec_district_arc_sr_cost
			rare_upkeep = @tec_district_arc_sr_upkeep
		}
		
		inline_script = districts/resources/tec_industrial_maintenance_upkeep
	}

	# Modifiers
		# Standard
			planet_modifier = {
				planet_housing_add = 5
			}
		# /
		# Planet
			triggered_planet_modifier = {	# Housing bonus
				potential = {
					has_planet_flag = tec_ecumenopolis_spec_admin	
				}
				modifier = {
					planet_housing_add = 1
				}
			}
		# /

		inline_script = districts/effects/tec_industrial_maintenance
	# /

	# Jobs
		inline_script = {
			script = jobs/tec_artisan
			count = 4
			mult = 1
			display = with_desc
			trigger = ""
		}

			script = jobs/tec_artisan
			count = 2
			mult = 1
			display = no_desc
			trigger = "
				has_planet_flag = tec_ecumenopolis_spec_admin
			"
		}
	# /
}
